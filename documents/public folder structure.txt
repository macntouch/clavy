-css
    -scss
        -components // стилизация компонентов, для каждого компонента свой файл, в названии каждое слово через тире
        -modules // вспомогательные миксины, функции для генерирования классов, свойств
        -partial // переиспользумые классы: button, margin, column и т. д.
        -style.css // подключение файлов

-fonts // шрифты, для каждого отдельная папка

-js //название каждого файла camelCase, кроме файлов компонентов
    -actions // действия над данными, изменение данных производится только через них
    -components // компоненты
    -constants // константы
    -libs // самописные утилитки, модули
    -mixins // расширения для компонетов (наследуются все методы)
    -pages // страницы, тоже самое что и компоненты
    -stores // хранилища, все данные берутся отсюда
    -app.js // корневой файл

-script.js //скомпиленные модули browserify
-style.css //скомпиленные стили sass
